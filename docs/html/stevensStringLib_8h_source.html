258 inline std::vector<std::string> separate( const std::string_view & str,
259 const char separator = ',',
260 const bool omitEmptyStrings = true )
261 {
262 //If we are given an empty string, just return an empty vector
263 if(str.empty())
264 {
265 return {};
266 }
267
268 //Initialize all the variables we'll be using in our loop
269 std::vector<std::string> separatedStringsVec;
270 std::string separatedString;
271 size_t currSeparatorIndex;
272 size_t prevSeparatorIndex = 0;
273
274 //Loop until we've found all of the separator chars in the string_view
275 while(true)
276 {
277 //Look for a separator char in the string_view
278 currSeparatorIndex = str.find(separator, prevSeparatorIndex);
279 //If the separator is not found, just get everything up until the end
280 if(currSeparatorIndex == std::string_view::npos)
281 {
282 separatedString = str.substr(prevSeparatorIndex, str.length());
283 separatedStringsVec.push_back(separatedString);
284 break;
285 }
286 //Once we've found the index, of the separator char, get all of the string that we jumped over since the last separator character
287 separatedString = str.substr(prevSeparatorIndex, currSeparatorIndex-prevSeparatorIndex);
288 //Push it bank into our vector
289 separatedStringsVec.push_back(separatedString);
290 //Set the prevSeparatorIndex to currSeparatorIndex now
291 prevSeparatorIndex = currSeparatorIndex + 1;
292 }
293
294 //If we are omitting empty strings in our result, erase all of that we found
295 if(omitEmptyStrings)
296 {
297 separatedStringsVec.erase(std::remove(separatedStringsVec.begin(), separatedStringsVec.end(), ""), separatedStringsVec.end());
298 }
299
300 return separatedStringsVec;

515 {
516 return false;

531 {
532 //Check to see if the first character in the string is a sign. We allow this!

534 if(str[i] == '-' || str[i] == '+')

536 continue;
537 }
538 default:
539 //Check to see if the character is a decimal point
540 if(str[i] == point)
541 {
542 if(!seenDecimalPoint)

544 seenDecimalPoint = true;
545 }
546 //Numbers in standard notation only contain a single decimal point, so we return false.
547 else
548 {
549 return false;
550 }
551 }
552 //Check to see if the character is a numerical digit
553 else if(!std::isdigit(str[i]))
554 {
555 //If we have a character that's not a numerical digit which is not a decimal point or negative sign, it's not in standard notation. Return false.
556 return false;
557 }

559 else if(str[i] != '0')
560 {
561 seenNonZeroDigit = true;
562 }

596 {
597 //If our string is empty, return false
598 if(str.empty())
599 {
600 return false;
601 }
602 //Get the decimal point for our locale
603 char point = std::use_facet< std::numpunct<char> >(std::cout.getloc()).decimal_point();
604 bool seenDecimalPoint = false;
605 //Read in the decimal part of the string
606 std::string decimalPart;
607 long long int endOfDecimalPartIndex = -1;
608 for(int i = 0; i < str.length(); i++)
609 {
610 //Check to see if we've found the character that indicates the end of our decimal part
611 if( str[i] == 'x' ||
612 str[i] == 'X' ||
613 str[i] == '*' ||
614 str[i] == 'e' ||
615 str[i] == 'E')
616 {
617 endOfDecimalPartIndex = i;
618 break;
619 }
620 decimalPart += str[i];
621 }
622 //If we got to the end of the string without assigning endOfDecimalPartIndex, return false, as there must be no exponent part.
623 if(endOfDecimalPartIndex == -1)

625 return false;
626 }
627 //Check for the validity of the decimal part
628 if(!isStandardNumber(decimalPart))
629 {
630 return false;
631 }
632 //Make sure that we haven't reached the end of the string already
633 if(endOfDecimalPartIndex == (str.length()-1))
634 {
635 //Looks like we only have a decimal part and an 'x','X','*','e', or an 'E'.
636 return false;
637 }
638 //Get the other part of the potential scientific number
639 std::string tenToThePowerPart;
640 tenToThePowerPart = str.substr(endOfDecimalPartIndex + 1);
641 //Based on the endOfDecimalPartChar, decide how to parse the tenToThePowerPart should be parsed
642 if( str[endOfDecimalPartIndex] == 'x' ||
643 str[endOfDecimalPartIndex] == 'X' ||
644 str[endOfDecimalPartIndex] == '*' )
645 {
646 //We are expecting to see a 10^ and an integer
647 if(tenToThePowerPart.length() < 4)
648 {
649 //The tenToThePower part is too short.
650 return false;
651 }
652 if(!stevensStringLib::startsWith(tenToThePowerPart, "10^"))
653 {
654 //We don't have a "10^" included in our tenToThePowerPart, so it can't be scientific notation.
655 return false;
656 }
657 //Now the rest of the tenToThePower part after 10^ must be an integer
658 if(!stevensStringLib::isInteger(tenToThePowerPart.substr(3)))
659 {
660 //The rest of the tenToThePower part is not a valid integer
661 return false;
662 }
663 }
664 else if( str[endOfDecimalPartIndex] == 'e' ||
665 str[endOfDecimalPartIndex] == 'E' )
666 {
667 //We are expecting to read an integer
668 if(!stevensStringLib::isInteger(tenToThePowerPart))
669 {
670 //The rest of the tenToThePower part is not a valid integer
671 return false;
672 }
673 }
674
675 //All checks have been passed. The number is in scientfic notation!
676 return true;
