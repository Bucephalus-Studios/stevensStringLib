82 std::vector<std::string> separate( const std::string & str,
83 const std::string & separator = ",",
84 const bool & omitEmptyStrings = true)
85 {
86 std::vector<std::string> separatedStrings = {};
87 std::string word = "";
88
89 //If we have a single character separator, we can use a more efficient method
90 if(separator.length() == 1)
91 {
92 //Taken from https://stackoverflow.com/a/5757851
93 std::istringstream split(str);
94 for (std::string each; std::getline(split, each, separator[0]); separatedStrings.push_back(each));
95 //Check to see if the last element needs a separator character trimmed off (if there exists a trailing separator character)
96 if(!separatedStrings.empty())
97 {
98 if(separatedStrings.back().back() == separator[0])
99 {
100 separatedStrings.back().pop_back();
101 }
102 }
103 }
104 else
105 {
106 //Iterate through the input string and piece together the strings we want to separate
107 for(int i = 0; i < str.length(); i++)
108 {
109 word += str[i];
110 //Every time we add a character to the word, we check to see if it contains the separator
111 if(contains(word, separator))
112 {
113 //Extract the separator from the word
114 word.erase(word.find(separator), separator.length());
115 //Push the word into the vector
116 separatedStrings.push_back(word);
117 //Clear the word
118 word.clear();
119 }
120 }
121 //If we had no separators at all, or if the string doesn't end in a separator, we push the remaining
122 //word to the vector.
123 separatedStrings.push_back(word);
124 }
125
126 //If we are omitting empty strings in our result, erase all of that we found
127 if(omitEmptyStrings)
128 {
129 separatedStrings.erase(std::remove(separatedStrings.begin(), separatedStrings.end(), ""), separatedStrings.end());
130 }
131
132 return separatedStrings;
133 }

188 bool isInteger( const std::string & str )
189 {
190 for (int charIndex = 0; charIndex < str.length(); charIndex++)
191 {
192 //The first index has the possibility of having a negative sign
193 if(charIndex == 0)
194 {
195 if(str[charIndex] == '-')
196 {
197 continue;
198 }
199 }
200 //Check to see if the character is a numerical digit
201 if (!isdigit(str[charIndex]))
202 {
203 return false;
204 }
205 }
206
207 //Now that we're sure we have a string that contains only digits and possibly a negative sign, we check
208 //to see if converting the string to an integer will cause errors
209 int value = 0;
210 const auto res = std::from_chars( str.data(),
211 str.data() + str.size(),
212 value );
213 if (res.ec == std::errc::invalid_argument)
214 {
215 //Invalid argument for std::from_chars()
216 return false;
217 }
218 else if (res.ec == std::errc::result_out_of_range)
219 {
220 //Overflow/underflow has occurred
221 return false;
222 }
223
224 //Otherwise, return true
225 return true;
226 }

236 bool isFloat( const std::string & str )
237 {
238 //Let's keep track of if we've seen the decimal point or not
239 bool seenDecimalPoint = false;
240 char point = std::use_facet< std::numpunct<char> >(std::cout.getloc()).decimal_point();
241 for (int charIndex = 0; charIndex < str.length(); charIndex++)
242 {
243 //The first index has the possibility of having a negative sign
244 if(charIndex == 0)
245 {
246 if(str[charIndex] == '-')
247 {
248 continue;
249 }
250 }
251 if(!seenDecimalPoint)
252 {
253 if(str[charIndex] == point)
254 {
255 seenDecimalPoint = true;
256 continue;
257 }
258 }
259 //Check to see if the character is a numerical digit
260 if(!isdigit(str[charIndex]))
261 {
262 return false;
263 }
264 }
265
266 //If we've looked at the whole string and not found a decimal point, then we couldn't be looking at a floating point number.
267 if(!seenDecimalPoint)
268 {
269 return false;
270 }
271
272 //Now that we're sure we have digits with a decimal point, and possibly a negative sign in front, we convert the string to
273 //a floating point number using std::from_chars to see if we run into any problems.
274 double value = 0;
275 const auto format = std::chars_format::general;
276 const auto res = std::from_chars( str.data(),
277 str.data() + str.size(),
278 value,
279 format );
280 if (res.ec == std::errc::invalid_argument)
281 {
282 //Invalid argument
283 return false;
284 }
285 else if (res.ec == std::errc::result_out_of_range)
286 {
287 //Overflow/unverflow has occurred
288 return false;
289 }
290
291 //Otherwise, return true
292 return true;
293 }

399 T mapifyStringHelper( T map,
400 std::string str,
401 std::string keyValueSeparator,
402 std::string pairSeparator,
403 bool ignoreWhitespace = true )
404 {
405 //Get rid of any whitespace if necessary (TODO: see if we can just remove whitespace between words-equals signs, and words-commas)
406 if(ignoreWhitespace)
407 {
408 str = removeWhitespace(str);
409 }
410
411 //Separate the pairs
412 std::vector<std::string> keysAndValues = separate(str,pairSeparator);
413
414 //Separate the keys and values
415 std::vector<std::string> keyAndValue;
416 for(int i = 0; i < keysAndValues.size(); i++)
417 {
418 keyAndValue = separate(keysAndValues[i], keyValueSeparator);
419 if(keyAndValue.size() == 0)
420 {
421 //No keys or values found, skip
422 continue;
423 }
424 else if(keyAndValue.size() == 1)
425 {
426 //Insert a key with an empty value
427 map[keyAndValue[0]] = "";
428 }
429 else if(keyAndValue.size() >= 2)
430 {
431 //Insert the key value pair into the map (only the first two elements)
432 map[keyAndValue[0]] = keyAndValue[1];
433 }
434 }
435
436 return map;
437 }

564 std::string wrapToWidth( std::string str,
565 int wrapWidth )
566 {
567 std::istringstream in(str);
568 std::string line;
569 std::string word;
570 std::string output = "";
571 int lineCutOffIndex = 0;
572 int numberOfLines = countLines(str);
573 int currLineNum = 0;
574
575 while(getline(in,line))
576 {
577 lineCutOffIndex = 0;
578 while(true)
579 {
580 //Check to see if we need to wrap this line around
581 if(line.length() > wrapWidth)
582 {
583 //Find the last space in the string that fits on the line
584 lineCutOffIndex = line.rfind(" ", wrapWidth);
585 //If we can't find a space in the string and the string is too long, we just cut the line and wrap to the next line
586 if(lineCutOffIndex == std::string::npos)
587 {
588 //Add as much of the line as we can to the output and then add a newline
589 output += line.substr(0, wrapWidth) + "\n";
590 //Continue looping until the rest of the line is added to the output
591 line = line.substr(wrapWidth+1);
592 }
593 //If we find a space...
594 else
595 {
596 //Add the part of the string before the cut off point to the output
597 output += line.substr(0, lineCutOffIndex) + "\n";
598 //Set the line equal to the cut off portion of the string and continue looping until the rest of the line is added to the output
599 line = line.substr(lineCutOffIndex+1);
600 }
601 }
602 else
603 {
604 output += line;
605 break;
606 }
607 }
608 currLineNum++;
609 //Add a new line when printing the next line from the string, or print a newline if we only have a newline character
610 if((currLineNum < numberOfLines) || ((line.length()) == 0 && (numberOfLines == 1)))
611 {
612 output += "\n";
613 }
614 }
615
616 return output;
617 }

705 std::string getWhitespaceString(const std::locale & loc)
706 {
707 std::string whitespace;
708 for (char ch = std::numeric_limits<char>::min(); ch < std::numeric_limits<char>::max(); ch++)
709 if (std::isspace(ch, loc))
710 whitespace += ch;
711 // to avoid infinte loop check char max outside the for loop.
712 if (std::isspace(std::numeric_limits<char>::max(), std::locale(loc)))
713 whitespace += std::numeric_limits<char>::max();
714 return whitespace;
715 }

728 {

731

733 if (strBegin == std::string::npos)
734 {
735 return ""; // no content
736 }
737
738 const auto strEnd = str.find_last_not_of(whitespace);
