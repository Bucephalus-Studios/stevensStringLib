258 const char separator = ',',
259 const bool omitEmptyStrings = true )
260 {
261 //If we are given an empty string, just return an empty vector
262 if(str.empty())
263 {
264 return {};
265 }
266
267 //Initialize all the variables we'll be using in our loop
268 std::vector<std::string> separatedStringsVec;
269 std::string separatedString;
270 size_t currSeparatorIndex;
271 size_t prevSeparatorIndex = 0;
272
273 //Loop until we've found all of the separator chars in the string_view
274 while(true)
275 {
276 //Look for a separator char in the string_view
277 currSeparatorIndex = str.find(separator, prevSeparatorIndex);
278 //If the separator is not found, just get everything up until the end
279 if(currSeparatorIndex == std::string_view::npos)
280 {
281 separatedString = str.substr(prevSeparatorIndex, str.length());
282 separatedStringsVec.push_back(separatedString);
283 break;
284 }
285 //Once we've found the index, of the separator char, get all of the string that we jumped over since the last separator character
286 separatedString = str.substr(prevSeparatorIndex, currSeparatorIndex-prevSeparatorIndex);
287 //Push it bank into our vector
288 separatedStringsVec.push_back(separatedString);
289 //Set the prevSeparatorIndex to currSeparatorIndex now
290 prevSeparatorIndex = currSeparatorIndex + 1;
291 }
292
293 //If we are omitting empty strings in our result, erase all of that we found
294 if(omitEmptyStrings)
295 {
296 separatedStringsVec.erase(std::remove(separatedStringsVec.begin(), separatedStringsVec.end(), ""), separatedStringsVec.end());
297 }

510 {

512 if(str.empty())
513 {
514 return false;
515 }
516 //If we have a string of length 1, just check to see if the single character is a digit

522 char point = std::use_facet< std::numpunct<char> >(std::cout.getloc()).decimal_point();
523 bool seenDecimalPoint = false;

525 //Otherwise, iterate through all of the characters in the string and examine each one

528 switch(i)
529 {
530 //Check to see if the first character in the string is a sign. We allow this!
531 case 0:
532 if(str[i] == '-' || str[i] == '+')
533 {
534 continue;
535 }
536 default:
537 //Check to see if the character is a decimal point
538 if(str[i] == point)
539 {
540 if(!seenDecimalPoint)
541 {
542 seenDecimalPoint = true;
543 }
544 //Numbers in standard notation only contain a single decimal point, so we return false.
545 else
546 {
547 return false;
548 }
549 }
550 //Check to see if the character is a numerical digit
551 else if(!std::isdigit(str[i]))
552 {
553 //If we have a character that's not a numerical digit which is not a decimal point or negative sign, it's not in standard notation. Return false.
554 return false;
555 }
556 //Check to see if the digit is nonzero
557 else if(str[i] != '0')
558 {
559 seenNonZeroDigit = true;
560 }
561 break;
562 }

568 return false;
569 }

572 return true;

593 {
594 //If our string is empty, return false

596 {
597 return false;

601 bool seenDecimalPoint = false;
602 //Read in the decimal part of the string
603 std::string decimalPart;
604 long long int endOfDecimalPartIndex = -1;
605 for(int i = 0; i < str.length(); i++)
606 {
607 //Check to see if we've found the character that indicates the end of our decimal part
608 if( str[i] == 'x' ||
609 str[i] == 'X' ||
610 str[i] == '*' ||
611 str[i] == 'e' ||
612 str[i] == 'E')

615 break;
616 }

623 }
624 //Check for the validity of the decimal part
625 if(!isStandardNumber(decimalPart))
626 {
627 return false;
628 }
629 //Make sure that we haven't reached the end of the string already
630 if(endOfDecimalPartIndex == (str.length()-1))
631 {
632 //Looks like we only have a decimal part and an 'x','X','*','e', or an 'E'.
633 return false;
634 }
635 //Get the other part of the potential scientific number
636 std::string tenToThePowerPart;
637 tenToThePowerPart = str.substr(endOfDecimalPartIndex + 1);
638 //Based on the endOfDecimalPartChar, decide how to parse the tenToThePowerPart should be parsed
639 if( str[endOfDecimalPartIndex] == 'x' ||
640 str[endOfDecimalPartIndex] == 'X' ||
641 str[endOfDecimalPartIndex] == '*' )
642 {
643 //We are expecting to see a 10^ and an integer
644 if(tenToThePowerPart.length() < 4)
645 {
646 //The tenToThePower part is too short.
647 return false;
648 }
649 if(!stevensStringLib::startsWith(tenToThePowerPart, "10^"))
650 {
651 //We don't have a "10^" included in our tenToThePowerPart, so it can't be scientific notation.
652 return false;
653 }
654 //Now the rest of the tenToThePower part after 10^ must be an integer
655 if(!stevensStringLib::isInteger(tenToThePowerPart.substr(3)))
656 {
657 //The rest of the tenToThePower part is not a valid integer
658 return false;
659 }
660 }
661 else if( str[endOfDecimalPartIndex] == 'e' ||
662 str[endOfDecimalPartIndex] == 'E' )
663 {
664 //We are expecting to read an integer
665 if(!stevensStringLib::isInteger(tenToThePowerPart))
666 {
667 //The rest of the tenToThePower part is not a valid integer
668 return false;
669 }
670 }
671
672 //All checks have been passed. The number is in scientfic notation!
673 return true;
