274 inline std::vector<std::string> separate( const std::string_view & str,
275 const char separator = ',',
276 const bool omitEmptyStrings = true )
277 {
278 //If we are given an empty string, just return an empty vector
279 if(str.empty())
280 {
281 return {};
282 }
283
284 //Initialize all the variables we'll be using in our loop
285 std::vector<std::string> separatedStringsVec;
286 std::string separatedString;
287 size_t currSeparatorIndex;
288 size_t prevSeparatorIndex = 0;
289
290 //Loop until we've found all of the separator chars in the string_view
291 while(true)
292 {
293 //Look for a separator char in the string_view
294 currSeparatorIndex = str.find(separator, prevSeparatorIndex);
295 //If the separator is not found, just get everything up until the end
296 if(currSeparatorIndex == std::string_view::npos)
297 {
298 separatedString = str.substr(prevSeparatorIndex, str.length());
299 separatedStringsVec.push_back(separatedString);
300 break;
301 }
302 //Once we've found the index, of the separator char, get all of the string that we jumped over since the last separator character
303 separatedString = str.substr(prevSeparatorIndex, currSeparatorIndex-prevSeparatorIndex);
304 //Push it bank into our vector
305 separatedStringsVec.push_back(separatedString);
306 //Set the prevSeparatorIndex to currSeparatorIndex now
307 prevSeparatorIndex = currSeparatorIndex + 1;
308 }
309
310 //If we are omitting empty strings in our result, erase all of that we found
311 if(omitEmptyStrings)
312 {
313 separatedStringsVec.erase(std::remove(separatedStringsVec.begin(), separatedStringsVec.end(), ""), separatedStringsVec.end());
314 }
315
316 return separatedStringsVec;

340 inline std::vector<std::string> separate( const std::string_view & str,
341 const std::string_view & separator,
342 const bool omitEmptyStrings = true )
343 {
344 //If we have a single character separator, we can use a more efficient method
345 if(separator.length() == 1)
346 {
347 return stevensStringLib::separate( str, separator[0], omitEmptyStrings );
348 }
349
350 //If separator is empty, split into individual characters
351 if(separator.empty())
352 {
353 std::vector<std::string> separatedStrings;
354 separatedStrings.reserve(str.length());
355 for(char c : str)
356 {
357 separatedStrings.emplace_back(1, c);
358 }
359 return separatedStrings;
360 }
361
362 std::vector<std::string> separatedStrings;
363 separatedStrings.reserve(8); // Reasonable default to avoid vector reallocations
364
365 size_t start = 0;
366 size_t pos = 0;
367
368 // Direct search without building intermediate strings
369 while((pos = str.find(separator, start)) != std::string_view::npos)
370 {
371 if(pos > start || !omitEmptyStrings)
372 {
373 // Construct string directly in vector memory - no temporary objects
374 separatedStrings.emplace_back(str.data() + start, pos - start);
375 }
376 start = pos + separator.length();
377 }
378
379 // Handle final segment
380 if(start < str.length() || (!omitEmptyStrings && start == str.length()))
381 {
382 separatedStrings.emplace_back(str.data() + start, str.length() - start);
383 }
384
385 return separatedStrings;

592 inline bool isStandardNumber( const std::string_view & str )
593 {
594 //If our string is empty, return false
595 if(str.empty())
596 {
597 return false;
598 }
599 //If we have a string of length 1, just check to see if the single character is a digit
600 else if(str.length() == 1)
601 {
602 return std::isdigit(str[0]);
603 }
604 //Get the decimal point for our locale
605 char point = std::use_facet< std::numpunct<char> >(std::cout.getloc()).decimal_point();
606 bool seenDecimalPoint = false;
607 bool seenNonZeroDigit = false;
608 //Otherwise, iterate through all of the characters in the string and examine each one
609 for(int i = 0; i < str.length(); i++)
610 {
611 switch(i)
612 {
613 //Check to see if the first character in the string is a sign. We allow this!
614 case 0:
615 if(str[i] == '-' || str[i] == '+')
616 {
617 continue;
618 }
619 default:
620 //Check to see if the character is a decimal point
621 if(str[i] == point)
622 {
623 if(!seenDecimalPoint)
624 {
625 seenDecimalPoint = true;
626 }
627 //Numbers in standard notation only contain a single decimal point, so we return false.
628 else
629 {
630 return false;
631 }
632 }
633 //Check to see if the character is a numerical digit
634 else if(!std::isdigit(str[i]))
635 {
636 //If we have a character that's not a numerical digit which is not a decimal point or negative sign, it's not in standard notation. Return false.
637 return false;
638 }
639 //Check to see if the digit is nonzero
640 else if(str[i] != '0')
641 {
642 seenNonZeroDigit = true;
643 }
644 break;
645 }
646 }
647
648 if((!seenNonZeroDigit) && (stevensStringLib::startsWith(str, "+") || stevensStringLib::startsWith(str, "-")))
649 {
650 //000.000 should not be written with a + or - sign out front in standard notation.
651 return false;
652 }
653
654 //We've made it past all of the checks by this point, so we must have a string representing a number in standard notation.
655 return true;
656 }

681 return false;

693 str[i] == 'X' ||
694 str[i] == '*' ||
695 str[i] == 'e' ||
696 str[i] == 'E')

699 break;

704 if(endOfDecimalPartIndex == -1)

706 return false;
707 }
708 //Check for the validity of the decimal part
709 if(!isStandardNumber(decimalPart))
710 {
711 return false;
712 }
713 //Make sure that we haven't reached the end of the string already
714 if(endOfDecimalPartIndex == (str.length()-1))
715 {
716 //Looks like we only have a decimal part and an 'x','X','*','e', or an 'E'.
717 return false;
718 }
719 //Get the other part of the potential scientific number
720 std::string tenToThePowerPart;
721 tenToThePowerPart = str.substr(endOfDecimalPartIndex + 1);
722 //Based on the endOfDecimalPartChar, decide how to parse the tenToThePowerPart should be parsed
723 if( str[endOfDecimalPartIndex] == 'x' ||
724 str[endOfDecimalPartIndex] == 'X' ||
725 str[endOfDecimalPartIndex] == '*' )
726 {
727 //We are expecting to see a 10^ and an integer
728 if(tenToThePowerPart.length() < 4)
729 {
730 //The tenToThePower part is too short.
731 return false;
732 }
733 if(!stevensStringLib::startsWith(tenToThePowerPart, "10^"))
734 {
735 //We don't have a "10^" included in our tenToThePowerPart, so it can't be scientific notation.
736 return false;
737 }
738 //Now the rest of the tenToThePower part after 10^ must be an integer
739 if(!stevensStringLib::isInteger(tenToThePowerPart.substr(3)))
740 {
741 //The rest of the tenToThePower part is not a valid integer
742 return false;
743 }
744 }
745 else if( str[endOfDecimalPartIndex] == 'e' ||
746 str[endOfDecimalPartIndex] == 'E' )
747 {
748 //We are expecting to read an integer
749 if(!stevensStringLib::isInteger(tenToThePowerPart))
750 {
751 //The rest of the tenToThePower part is not a valid integer
752 return false;
753 }
754 }
755
756 //All checks have been passed. The number is in scientific notation!
757 return true;
758 }
