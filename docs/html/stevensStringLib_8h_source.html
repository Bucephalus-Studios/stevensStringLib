73 std::vector<std::string> separate( const std::string & input,
74 const std::string & separator = ",",
75 const bool & omitEmptyStrings = true)
76 {
77 std::vector<std::string> separatedStrings = {};
78 std::string word = "";
79
80 //If we have a single character separator, we can use a more efficient method
81 if(separator.length() == 1)
82 {
83 //Taken from https://stackoverflow.com/a/5757851
84 std::istringstream split(input);
85 for (std::string each; std::getline(split, each, separator[0]); separatedStrings.push_back(each));
86 //Check to see if the last element needs a separator character trimmed off (if there exists a trailing separator character)
87 if(!separatedStrings.empty())
88 {
89 if(separatedStrings.back().back() == separator[0])
90 {
91 separatedStrings.back().pop_back();
92 }
93 }
94 }
95 else
96 {
97 //Iterate through the input string and piece together the strings we want to separate
98 for(int i = 0; i < input.length(); i++)
99 {
100 word += input[i];
101 //Every time we add a character to the word, we check to see if it contains the separator
102 if(contains(word, separator))
103 {
104 //Extract the separator from the word
105 word.erase(word.find(separator), separator.length());
106 //Push the word into the vector
107 separatedStrings.push_back(word);
108 //Clear the word
109 word.clear();
110 }
111 }
112 //If we had no separators at all, or if the string doesn't end in a separator, we push the remaining
113 //word to the vector.
114 separatedStrings.push_back(word);
115 }
116
117 //If we are omitting empty strings in our result, erase all of that we found
118 if(omitEmptyStrings)
119 {
120 separatedStrings.erase(std::remove(separatedStrings.begin(), separatedStrings.end(), ""), separatedStrings.end());
121 }
122
123 return separatedStrings;
124 }

175 bool isInteger( const std::string & str )
176 {
177 for (int charIndex = 0; charIndex < str.length(); charIndex++)
178 {
179 //The first index has the possibility of having a negative sign
180 if(charIndex == 0)
181 {
182 if(str[charIndex] == '-')
183 {
184 continue;
185 }
186 }
187 //Check to see if the character is a numerical digit
188 if (!isdigit(str[charIndex]))
189 {
190 return false;
191 }
192 }
193
194 //Now that we're sure we have a string that contains only digits and possibly a negative sign, we check
195 //to see if converting the string to an integer will cause errors
196 int value = 0;
197 const auto res = std::from_chars( str.data(),
198 str.data() + str.size(),
199 value );
200 if (res.ec == std::errc::invalid_argument)
201 {
202 //Invalid argument for std::from_chars()
203 return false;
204 }
205 else if (res.ec == std::errc::result_out_of_range)
206 {
207 //Overflow/underflow has occurred
208 return false;
209 }
210
211 //Otherwise, return true
212 return true;
213 }

225 bool isFloat( const std::string & str )
226 {
227 //Let's keep track of if we've seen the decimal point or not
228 bool seenDecimalPoint = false;
229 char point = std::use_facet< std::numpunct<char> >(std::cout.getloc()).decimal_point();
230 for (int charIndex = 0; charIndex < str.length(); charIndex++)
231 {
232 //The first index has the possibility of having a negative sign
233 if(charIndex == 0)
234 {
235 if(str[charIndex] == '-')
236 {
237 continue;
238 }
239 }
240 if(!seenDecimalPoint)
241 {
242 if(str[charIndex] == point)
243 {
244 seenDecimalPoint = true;
245 continue;
246 }
247 }
248 //Check to see if the character is a numerical digit
249 if(!isdigit(str[charIndex]))
250 {
251 return false;
252 }
253 }
254
255 //If we've looked at the whole string and not found a decimal point, then we couldn't be looking at a floating point number.
256 if(!seenDecimalPoint)
257 {
258 return false;
259 }
260
261 //Now that we're sure we have digits with a decimal point, and possibly a negative sign in front, we convert the string to
262 //a floating point number using std::from_chars to see if we run into any problems.
263 double value = 0;
264 const auto format = std::chars_format::general;
265 const auto res = std::from_chars( str.data(),
266 str.data() + str.size(),
267 value,
268 format );
269 if (res.ec == std::errc::invalid_argument)
270 {
271 //Invalid argument
272 return false;
273 }
274 else if (res.ec == std::errc::result_out_of_range)
275 {
276 //Overflow/unverflow has occurred
277 return false;
278 }
279
280 //Otherwise, return true
281 return true;
282 }

402 std::map<std::string,std::string> mapifyString( std::string str,
403 std::string keyValueSeparator,
404 std::string pairSeparator,
405 bool ignoreWhitespace = true )
406 {
407 //Create the map that we'll be returning later
408 std::map<std::string,std::string> mappedString = {};
409
410 //Get rid of any whitespace if necessary (TODO: see if we can just remove whitespace between words-equals signs, and words-commas)
411 if(ignoreWhitespace)
412 {
413 str = removeWhitespace(str);
414 }
415
416 //Separate the pairs
417 std::vector<std::string> keysAndValues = separate(str,pairSeparator);
418
419 //Separate the keys and values
420 std::vector<std::string> keyAndValue;
421 for(int i = 0; i < keysAndValues.size(); i++)
422 {
423 keyAndValue = separate(keysAndValues[i], keyValueSeparator);
424 if(keyAndValue.size() == 0)
425 {
426 //No keys or values found, skip
427 continue;
428 }
429 else if(keyAndValue.size() == 1)
430 {
431 //Insert a key with an empty value
432 mappedString[keyAndValue[0]] = "";
433 }
434 else if(keyAndValue.size() >= 2)
435 {
436 //Insert the key value pair into the map (only the first two elements)
437 mappedString[keyAndValue[0]] = keyAndValue[1];
438 }
439 }
440
441 return mappedString;
442 }

589 std::string wrapToWidth( std::string str,
590 int wrapWidth )
591 {
592 std::istringstream in(str);
593 std::string line;
594 std::string word;
595 std::string output = "";
596 int lineCutOffIndex = 0;
597 int numberOfLines = countLines(str);
598 int currLineNum = 0;
599
600 while(getline(in,line))
601 {
602 lineCutOffIndex = 0;
603 while(true)
604 {
605 //Check to see if we need to wrap this line around
606 if(line.length() > wrapWidth)
607 {
608 //Find the last space in the string that fits on the line
609 lineCutOffIndex = line.rfind(" ", wrapWidth);
610 //If we can't find a space in the string and the string is too long, we just cut the line and wrap to the next line
611 if(lineCutOffIndex == std::string::npos)
612 {
613 //Add as much of the line as we can to the output and then add a newline
614 output += line.substr(0, wrapWidth) + "\n";
615 //Continue looping until the rest of the line is added to the output
616 line = line.substr(wrapWidth+1);
617 }
618 //If we find a space...
619 else
620 {
621 //Add the part of the string before the cut off point to the output
622 output += line.substr(0, lineCutOffIndex) + "\n";
623 //Set the line equal to the cut off portion of the string and continue looping until the rest of the line is added to the output
624 line = line.substr(lineCutOffIndex+1);
625 }
626 }
627 else
628 {
629 output += line;
630 break;
631 }
632 }
633 currLineNum++;
634 //Add a new line when printing the next line from the string, or print a newline if we only have a newline character
635 if((currLineNum < numberOfLines) || ((line.length()) == 0 && (numberOfLines == 1)))
636 {
637 output += "\n";
638 }
639 }
640
641 return output;
642 }

744 std::string getWhitespaceString(const std::locale & loc)
745 {
746 std::string whitespace;
747 for (char ch = std::numeric_limits<char>::min(); ch < std::numeric_limits<char>::max(); ch++)
748 if (std::isspace(ch, loc))
749 whitespace += ch;
750 // to avoid infinte loop check char max outside the for loop.
751 if (std::isspace(std::numeric_limits<char>::max(), std::locale(loc)))
752 whitespace += std::numeric_limits<char>::max();
753 return whitespace;
754 }

768 std::string trimWhitespace( std::string str )
769 {

772

774 if (strBegin == std::string::npos)

777 }
778
779 const auto strEnd = str.find_last_not_of(whitespace);
